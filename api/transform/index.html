class Compose(Transform):
    """Compose class."""

    t: Transform
    u: Transform

    def __call__(self) -> cairo.Matrix:
        # return self.t().multiply(self.u())
        return self.u().multiply(self.t())

    def to_svg(self) -> str:
        return self.t.to_svg() + " " + self.u.to_svg()

        return self.apply_transform(Rotate(Œ∏))
